function paginadorInit(startPage,perPage,containerID,paginadorClass){
	minHeight = false;
    $(paginadorClass).jPages({
        containerID  : containerID,
        perPage      : perPage,
        startPage    : startPage,
        startRange   : 1,
        midRange     : 5,
        endRange     : 1,
        first        : '',
        previous     : 'anterior',
        next         : 'siguiente',
        last         : '',
        minHeight    : minHeight,
        callback     : function(pages,items){
                       }
    });    
}
